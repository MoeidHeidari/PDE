120 /*Calculate the integral X= a + (i + 0.5) * step*/
121 double integral(double(*f)(double x_val), double(*g)(double x_val), double a, double b, int n)
122 {
123  double step = (b - a)/n; // width of rectangle
124  double area = 0.0;
125  double y = 0; // height of rectangle
126  for(int i = 0; i < n; ++i)
127  {
128 
129  y = f(a + (i + 0.5) * step) * g((a + (i + 0.5) * step)*PDE_PI*n);
130 
131  area += y * step; // find the area of the rectangle and add it to the previous area. Effectively summing up the area under the curve.
132 
133  }
134 
135 
136  return area;
137 }
