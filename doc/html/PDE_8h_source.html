190  double integral(double(*f)(double x_val), double(*g)(double x_val), double a, double b, int n)
191  {

193  double area = 0.0;
194  double y = 0;

196  {
197  y = f(a + (i + 0.5) * step) * g((a + (i + 0.5) * step)*PDE_PI*n);
